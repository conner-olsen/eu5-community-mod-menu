# Root scope: country.
cmm_example2_register_country = {
	cmm_register_mod = {
		mod_id = cmm_example2
	}

	cmm_register_bool_setting = {
		mod_id = cmm_example2
		setting_id = ai_helper
		default_value = 0
	}

	cmm_register_bool_setting = {
		mod_id = cmm_example2
		setting_id = map_overlay
		default_value = 1
	}
}

cmm_example2__ai_helper_on_changed = {
	if = {
		limit = {
			var:$setting$ = 1
		}
		set_variable = { name = cmm_example2_ai_helper_enabled value = 1 }
	}
	else = {
		set_variable = { name = cmm_example2_ai_helper_enabled value = 0 }
	}
}

cmm_example2__map_overlay_on_changed = {
	if = {
		limit = {
			var:$setting$ = 1
		}
		set_variable = { name = cmm_example2_map_overlay_enabled value = 1 }
	}
	else = {
		set_variable = { name = cmm_example2_map_overlay_enabled value = 0 }
	}
}
