cmm_example__feature_enabled_on_changed = {
	scope = country

	effect = {
		cmm_toggle_bool_setting = {
			setting = cmm_example__feature_enabled
		}

		if = {
			limit = { var:cmm_example__feature_enabled = 1 }
			# enabled effect
		}
		else = {
			# disabled effect
		}
	}
}

cmm_example__monthly_bonus_on_changed = {
	scope = country

	effect = {
		set_variable = { name = cmm_example__monthly_bonus value = var:cmm_example__monthly_bonus }
	}
}

cmm_example__monthly_bonus_on_decrease = {
	scope = country

	effect = {
		cmm_step_numeric_setting_down = {
			setting = cmm_example__monthly_bonus
		}
	}
}

cmm_example__monthly_bonus_on_increase = {
	scope = country

	effect = {
		cmm_step_numeric_setting_up = {
			setting = cmm_example__monthly_bonus
		}
	}
}
