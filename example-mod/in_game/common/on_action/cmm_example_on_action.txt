# Country-scope hooks for registering this mod with Community Mod Menu.
on_game_start = {
    on_actions = {
        delay = { days = 1 }
        cmm_example_register_all_humans
    }
}

yearly_country_pulse = {
    on_actions = { cmm_example_ensure_registered }
}

cmm_example_register_all_humans = {
    effect = {
        every_country = {
            limit = { is_ai = no }
            cmm_example_register_country = yes
            cmm_example_consume_cmm_dirty = yes
        }
    }
}

cmm_example_ensure_registered = {
    trigger = { is_ai = no }
    effect = {
        cmm_example_register_country = yes
        cmm_example_consume_cmm_dirty = yes
    }
}
