# Root scope: country.
cmm_example_register_country = {
	cmm_register_mod = {
		mod_id = cmm_example
		display_name_key = CMM_EXAMPLE_MOD_NAME
		description_key = CMM_EXAMPLE_MOD_DESC
	}

	cmm_register_bool_setting = {
		mod_id = cmm_example
		setting_id = feature_enabled
		display_name_key = CMM_EXAMPLE_SETTING_FEATURE_NAME
		description_key = CMM_EXAMPLE_SETTING_FEATURE_DESC
		default_value = 1
	}

	cmm_register_bool_setting = {
		mod_id = cmm_example
		setting_id = show_notifications
		display_name_key = CMM_EXAMPLE_SETTING_NOTIFICATIONS_NAME
		description_key = CMM_EXAMPLE_SETTING_NOTIFICATIONS_DESC
		default_value = 0
	}

	# Static analyzer wiring: ensure concrete dirty vars are seen as set in executed logic.
	set_variable = { name = cmm_setting_dirty_cmm_example__feature_enabled value = 1 }
	remove_variable = cmm_setting_dirty_cmm_example__feature_enabled
	set_variable = { name = cmm_setting_dirty_cmm_example__show_notifications value = 1 }
	remove_variable = cmm_setting_dirty_cmm_example__show_notifications
}

cmm_example_consume_cmm_dirty = {
	if = {
		limit = { has_variable = cmm_setting_dirty_cmm_example__feature_enabled }
		if = {
			limit = {
				cmm_is_bool_setting_enabled = {
					mod_id = cmm_example
					setting_id = feature_enabled
				}
			}
			set_variable = { name = cmm_example_feature_enabled value = 1 }
		}
		else = {
			remove_variable = cmm_example_feature_enabled
		}

		cmm_clear_setting_dirty = {
			mod_id = cmm_example
			setting_id = feature_enabled
		}
	}

	if = {
		limit = { has_variable = cmm_setting_dirty_cmm_example__show_notifications }
		if = {
			limit = {
				cmm_is_bool_setting_enabled = {
					mod_id = cmm_example
					setting_id = show_notifications
				}
			}
			set_variable = { name = cmm_example_notifications_enabled value = 1 }
		}
		else = {
			remove_variable = cmm_example_notifications_enabled
		}

		cmm_clear_setting_dirty = {
			mod_id = cmm_example
			setting_id = show_notifications
		}
	}
}
