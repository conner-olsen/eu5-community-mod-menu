# Queue a one-step numeric decrease.
# Scope: country
# Args: none
CMM_MarkNumericDecrease = {
    effect = {
        if = {
            limit = { has_variable = cmm_numeric_step_delta_to_apply }
            change_variable = { name = cmm_numeric_step_delta_to_apply add = -1 }
        }
        else = {
            set_variable = { name = cmm_numeric_step_delta_to_apply value = -1 }
        }

        if = {
            limit = { has_variable = cmm_numeric_change_to_apply }
            remove_variable = cmm_numeric_change_to_apply
        }
    }
}
