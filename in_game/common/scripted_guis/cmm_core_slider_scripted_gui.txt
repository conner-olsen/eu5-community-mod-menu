# Begin a slider drag interaction.
# Scope: country
# Args: none
CMM_StartSliderDrag = {
	effect = {
		set_variable = { name = cmm_slider_dragging value = 1 }
	}
}

# End a slider drag interaction.
# Scope: country
# Args: none
CMM_StopSliderDrag = {
	effect = {
		if = {
			limit = { has_variable = cmm_slider_dragging }
			remove_variable = cmm_slider_dragging
		}
	}
}
