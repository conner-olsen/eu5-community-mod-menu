# Register core CMM metadata and run the registration hook.
# Scope: scripted_gui root
# Args: none
CMM_RegisterCoreMod = {
	effect = {
		cmm_register_core_country = yes
		trigger_event_silently = {
			on_action = cmm_on_register_country
		}
	}
}

# Mark the next numeric interaction as decrease.
# Scope: scripted_gui root
# Args: none
CMM_MarkNumericDecrease = {
	effect = {
		set_variable = { name = cmm_change_to_apply value = 0 }
	}
}

# Mark the next numeric interaction as increase.
# Scope: scripted_gui root
# Args: none
CMM_MarkNumericIncrease = {
	effect = {
		set_variable = { name = cmm_change_to_apply value = 1 }
	}
}
