# Run built-in CMM registration and the shared registration hook.
# Scope: country
# Args: none
CMM_RegisterCoreMod = {
	effect = {
		cmm_register_menu_settings = yes
		trigger_event_silently = {
			on_action = cmm_on_mod_registration
		}
	}
}

# Mark the clicking country as the CMM host country.
# Scope: country
# Args: none
cmm_core__mark_host_country_on_changed = {
	scope = country
	effect = {
		cmm_mark_host_country = yes
	}
}
