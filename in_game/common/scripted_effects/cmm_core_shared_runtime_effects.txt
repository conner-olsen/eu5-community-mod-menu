# Toggle a registered boolean setting value between `0` and `1`.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_toggle_bool_setting = {
	if = {
		limit = { var:cmm_setting_is_global_$setting$ = 1 }
		if = {
			limit = { global_var:$setting$ = 1 }
			set_global_variable = { name = $setting$ value = 0 }
		}
		else = {
			set_global_variable = { name = $setting$ value = 1 }
		}
	}
	else = {
		if = {
			limit = { var:$setting$ = 1 }
			set_variable = { name = $setting$ value = 0 }
		}
		else = {
			set_variable = { name = $setting$ value = 1 }
		}
	}
}

# Apply the pending dropdown selected index to a setting and clear the marker.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_apply_dropdown_change = {
	if = {
		limit = { has_variable = cmm_dropdown_index_to_apply }

		if = {
			limit = { var:cmm_setting_is_global_$setting$ = 1 }
			set_global_variable = { name = $setting$ value = var:cmm_dropdown_index_to_apply }

			if = {
				limit = { global_var:$setting$ < 0 }
				set_global_variable = { name = $setting$ value = 0 }
			}

			if = {
				limit = { global_var:$setting$ > var:cmm_setting_dropdown_last_index_$setting$ }
				set_global_variable = { name = $setting$ value = var:cmm_setting_dropdown_last_index_$setting$ }
			}
		}
		else = {
			set_variable = { name = $setting$ value = var:cmm_dropdown_index_to_apply }

			if = {
				limit = { var:$setting$ < 0 }
				set_variable = { name = $setting$ value = 0 }
			}

			if = {
				limit = { var:$setting$ > var:cmm_setting_dropdown_last_index_$setting$ }
				set_variable = { name = $setting$ value = var:cmm_setting_dropdown_last_index_$setting$ }
			}
		}

		remove_variable = cmm_dropdown_index_to_apply
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = var:cmm_dropdown_index_to_apply
			exists = var:cmm_setting_dropdown_last_index_$setting$
		}
	}
}
