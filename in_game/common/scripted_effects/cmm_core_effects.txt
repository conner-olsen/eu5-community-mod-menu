# Register a mod (idempotent by mod_id).
# Scope: country
# Args:
# - mod_id (key): Mod identifier; derives `<mod_id>_name` and `<mod_id>_desc`.
cmm_register_mod = {
	# First-time registration path for this mod id.
	if = {
		limit = { NOT = { has_variable = cmm_registered_mod_$mod_id$ } }
		set_variable = { name = cmm_registered_mod_$mod_id$ value = yes }
		set_variable = { name = $mod_id$_name value = flag:$mod_id$_name }
		set_variable = { name = $mod_id$_desc value = flag:$mod_id$_desc }

		# Global mod id list; used by left-panel datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_mod_ids
			target = flag:$mod_id$
		}
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$
			exists = flag:$mod_id$_name
			exists = flag:$mod_id$_desc
			exists = var:cmm_registered_mod_ids
			exists = var:cmm_registered_mod_$mod_id$
			exists = var:$mod_id$_name
			exists = var:$mod_id$_desc
		}
	}
}

# Register a tab for a mod (idempotent by mod_id + tab_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - tab_id (key): Tab id unique within mod.
cmm_register_tab = {
	# First-time registration path for this tab key.
	if = {
		limit = { NOT = { has_variable = cmm_registered_tab_$mod_id$__$tab_id$ } }
		set_variable = { name = cmm_registered_tab_$mod_id$__$tab_id$ value = yes }
		set_variable = { name = cmm_tab_owner_mod_id_$mod_id$__$tab_id$ value = flag:$mod_id$ }
		set_variable = { name = $mod_id$__$tab_id$_name value = flag:$mod_id$_$tab_id$_name }

		# Global tab key list; used by right-panel tab datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_tab_keys
			target = flag:$mod_id$__$tab_id$
		}

		# First registered tab becomes default selected tab for this mod.
		if = {
			limit = { NOT = { has_variable = cmm_mod_default_tab_$mod_id$ } }
			set_variable = { name = cmm_mod_default_tab_$mod_id$ value = flag:$mod_id$__$tab_id$ }
		}
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$__$tab_id$
			exists = flag:$mod_id$_$tab_id$_name
			exists = var:cmm_registered_tab_keys
			exists = var:cmm_registered_tab_$mod_id$__$tab_id$
			exists = var:cmm_mod_default_tab_$mod_id$
			exists = var:cmm_tab_owner_mod_id_$mod_id$__$tab_id$
			exists = var:$mod_id$__$tab_id$_name
		}
	}
}

# Register a boolean setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - default_value (1/0): Initial value; applied when the setting does not exist yet.
cmm_register_bool_setting = {
	# Ensure owning tab exists before binding setting metadata.
	cmm_register_tab = {
		mod_id = $mod_id$
		tab_id = $tab_id$
	}

	# First-time registration path for this setting key.
	if = {
		limit = { NOT = { has_variable = cmm_registered_setting_$mod_id$__$setting_id$ } }
		set_variable = { name = cmm_registered_setting_$mod_id$__$setting_id$ value = yes }
		set_variable = { name = cmm_setting_owner_mod_id_$mod_id$__$setting_id$ value = flag:$mod_id$ }
		set_variable = { name = cmm_setting_owner_tab_key_$mod_id$__$setting_id$ value = flag:$mod_id$__$tab_id$ }
		set_variable = { name = $mod_id$__$setting_id$_name value = flag:$mod_id$_$setting_id$_name }
		set_variable = { name = $mod_id$__$setting_id$_desc value = flag:$mod_id$_$setting_id$_desc }

		# Global setting key list; used by right-panel datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_setting_keys
			target = flag:$mod_id$__$setting_id$
		}

		# Per-tab count; used for "no settings in this tab" UI state.
		if = {
			limit = { NOT = { has_variable = cmm_tab_setting_count_$mod_id$__$tab_id$ } }
			set_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ value = 0 }
		}
		change_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ add = 1 }
	}

	# Default initialization; preserves existing save value.
	if = {
		limit = { NOT = { has_variable = $mod_id$__$setting_id$ } }
		set_variable = { name = $mod_id$__$setting_id$ value = $default_value$ }
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$__$setting_id$
			exists = flag:$mod_id$__$tab_id$
			exists = flag:$mod_id$_$setting_id$_name
			exists = flag:$mod_id$_$setting_id$_desc
			exists = var:cmm_registered_setting_keys
			exists = var:cmm_setting_owner_mod_id_$mod_id$__$setting_id$
			exists = var:cmm_setting_owner_tab_key_$mod_id$__$setting_id$
			exists = var:cmm_tab_setting_count_$mod_id$__$tab_id$
			exists = var:$mod_id$__$setting_id$_name
			exists = var:$mod_id$__$setting_id$_desc
		}
	}
}

# Register a numeric setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - default_value (number): Initial value; applied when the setting does not exist yet.
# - min_value (number): Lower clamp bound.
# - max_value (number): Upper clamp bound.
# - step_value (number): Increase/decrease step size.
cmm_register_numeric_setting = {
	# Ensure owning tab exists before binding setting metadata.
	cmm_register_tab = {
		mod_id = $mod_id$
		tab_id = $tab_id$
	}

	set_variable = { name = cmm_setting_min_$mod_id$__$setting_id$ value = $min_value$ }
	set_variable = { name = cmm_setting_max_$mod_id$__$setting_id$ value = $max_value$ }
	set_variable = { name = cmm_setting_step_$mod_id$__$setting_id$ value = $step_value$ }

	# First-time registration path for this setting key.
	if = {
		limit = { NOT = { has_variable = cmm_registered_setting_$mod_id$__$setting_id$ } }
		set_variable = { name = cmm_registered_setting_$mod_id$__$setting_id$ value = yes }
		set_variable = { name = cmm_setting_owner_mod_id_$mod_id$__$setting_id$ value = flag:$mod_id$ }
		set_variable = { name = cmm_setting_owner_tab_key_$mod_id$__$setting_id$ value = flag:$mod_id$__$tab_id$ }
		set_variable = { name = $mod_id$__$setting_id$_name value = flag:$mod_id$_$setting_id$_name }
		set_variable = { name = $mod_id$__$setting_id$_desc value = flag:$mod_id$_$setting_id$_desc }

		# Global setting key list; used by right-panel datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_setting_keys
			target = flag:$mod_id$__$setting_id$
		}

		# Per-tab count; used for "no settings in this tab" UI state.
		if = {
			limit = { NOT = { has_variable = cmm_tab_setting_count_$mod_id$__$tab_id$ } }
			set_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ value = 0 }
		}
		change_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ add = 1 }
	}

	# Default initialization; preserves existing save value.
	if = {
		limit = { NOT = { has_variable = $mod_id$__$setting_id$ } }
		set_variable = { name = $mod_id$__$setting_id$ value = $default_value$ }
	}

	# Clamp current value into configured range.
	if = {
		limit = { var:$mod_id$__$setting_id$ < $min_value$ }
		set_variable = { name = $mod_id$__$setting_id$ value = $min_value$ }
	}

	if = {
		limit = { var:$mod_id$__$setting_id$ > $max_value$ }
		set_variable = { name = $mod_id$__$setting_id$ value = $max_value$ }
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$__$setting_id$
			exists = flag:$mod_id$__$tab_id$
			exists = flag:$mod_id$_$setting_id$_name
			exists = flag:$mod_id$_$setting_id$_desc
			exists = var:cmm_registered_setting_keys
			exists = var:cmm_setting_owner_mod_id_$mod_id$__$setting_id$
			exists = var:cmm_setting_owner_tab_key_$mod_id$__$setting_id$
			exists = var:cmm_setting_min_$mod_id$__$setting_id$
			exists = var:cmm_setting_max_$mod_id$__$setting_id$
			exists = var:cmm_setting_step_$mod_id$__$setting_id$
			exists = var:cmm_tab_setting_count_$mod_id$__$tab_id$
			exists = var:$mod_id$__$setting_id$_name
			exists = var:$mod_id$__$setting_id$_desc
		}
	}
}

# Toggle a registered boolean setting value between `0` and `1`.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_toggle_bool_setting = {
	if = {
		limit = { var:$setting$ = 1 }
		set_variable = { name = $setting$ value = 0 }
	}
	else = {
		set_variable = { name = $setting$ value = 1 }
	}
}

# Increase a numeric setting and clamp to its max value.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_step_numeric_setting_up = {
	change_variable = { name = $setting$ add = var:cmm_setting_step_$setting$ }

	if = {
		limit = { var:$setting$ > var:cmm_setting_max_$setting$ }
		set_variable = { name = $setting$ value = var:cmm_setting_max_$setting$ }
	}
}

# Increase a numeric setting by five intervals and clamp to max.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_step_numeric_setting_up_5x = {
	cmm_step_numeric_setting_up = { setting = $setting$ }
	cmm_step_numeric_setting_up = { setting = $setting$ }
	cmm_step_numeric_setting_up = { setting = $setting$ }
	cmm_step_numeric_setting_up = { setting = $setting$ }
	cmm_step_numeric_setting_up = { setting = $setting$ }
}

# Apply the pending numeric change marker to a setting and clear the marker.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_apply_numeric_change = {
	if = {
		limit = { has_variable = cmm_numeric_change_to_apply }

		switch = {
			trigger = var:cmm_numeric_change_to_apply
			1 = { cmm_step_numeric_setting_up = { setting = $setting$ } }
			2 = { cmm_step_numeric_setting_up_5x = { setting = $setting$ } }
			3 = { cmm_set_numeric_setting_max = { setting = $setting$ } }
			-1 = { cmm_step_numeric_setting_down = { setting = $setting$ } }
			-2 = { cmm_step_numeric_setting_down_5x = { setting = $setting$ } }
			-3 = { cmm_set_numeric_setting_min = { setting = $setting$ } }
		}

		remove_variable = cmm_numeric_change_to_apply
	}
}
