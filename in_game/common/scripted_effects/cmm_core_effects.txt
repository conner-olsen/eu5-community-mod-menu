# Register a mod (idempotent by mod_id).
# Scope: country
# Args:
# - mod_id (key): Mod identifier; derives `<mod_id>_name` and `<mod_id>_desc`.
cmm_register_mod = {
	# First-time registration path for this mod id.
	if = {
		limit = { NOT = { has_variable = cmm_registered_mod_$mod_id$ } }
		set_variable = { name = cmm_registered_mod_$mod_id$ value = yes }
		set_variable = { name = $mod_id$_name value = flag:$mod_id$_name }
		set_variable = { name = $mod_id$_desc value = flag:$mod_id$_desc }

		# Global mod id list; used by left-panel datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_mod_ids
			target = flag:$mod_id$
		}
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$
			exists = flag:$mod_id$_name
			exists = flag:$mod_id$_desc
			exists = var:cmm_registered_mod_ids
			exists = var:cmm_registered_mod_$mod_id$
			exists = var:$mod_id$_name
			exists = var:$mod_id$_desc
		}
	}
}

# Register a tab for a mod (idempotent by mod_id + tab_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - tab_id (key): Tab id unique within mod.
cmm_register_tab = {
	# First-time registration path for this tab key.
	if = {
		limit = { NOT = { has_variable = cmm_registered_tab_$mod_id$__$tab_id$ } }
		set_variable = { name = cmm_registered_tab_$mod_id$__$tab_id$ value = yes }
		set_variable = { name = cmm_tab_owner_mod_id_$mod_id$__$tab_id$ value = flag:$mod_id$ }
		set_variable = { name = $mod_id$__$tab_id$_name value = flag:$mod_id$_$tab_id$_name }

		# Global tab key list; used by right-panel tab datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_tab_keys
			target = flag:$mod_id$__$tab_id$
		}

		# First registered tab becomes default selected tab for this mod.
		if = {
			limit = { NOT = { has_variable = cmm_mod_default_tab_$mod_id$ } }
			set_variable = { name = cmm_mod_default_tab_$mod_id$ value = flag:$mod_id$__$tab_id$ }
		}
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$__$tab_id$
			exists = flag:$mod_id$_$tab_id$_name
			exists = var:cmm_registered_tab_keys
			exists = var:cmm_registered_tab_$mod_id$__$tab_id$
			exists = var:cmm_mod_default_tab_$mod_id$
			exists = var:cmm_tab_owner_mod_id_$mod_id$__$tab_id$
			exists = var:$mod_id$__$tab_id$_name
		}
	}
}

# Register a boolean setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:

# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - default_value (1/0): Initial value; applied when the setting does not exist yet.
cmm_register_bool_setting = {
	# Ensure owning tab exists before binding setting metadata.
	cmm_register_tab = {
		mod_id = $mod_id$
		tab_id = $tab_id$
	}

	set_variable = { name = cmm_setting_is_global_$mod_id$__$setting_id$ value = 0 }
	set_variable = { name = cmm_setting_is_slider_$mod_id$__$setting_id$ value = 0 }

	# First-time registration path for this setting key.

		set_variable = { name = cmm_setting_owner_mod_id_$mod_id$__$setting_id$ value = flag:$mod_id$ }
		set_variable = { name = cmm_setting_owner_tab_key_$mod_id$__$setting_id$ value = flag:$mod_id$__$tab_id$ }

		set_variable = { name = $mod_id$__$setting_id$_desc value = flag:$mod_id$_$setting_id$_desc }

		# Global setting key list; used by right-panel datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_setting_keys
			target = flag:$mod_id$__$setting_id$
		}

		# Per-tab count; used for "no settings in this tab" UI state.
		if = {
			limit = { NOT = { has_variable = cmm_tab_setting_count_$mod_id$__$tab_id$ } }
			set_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ value = 0 }
		}
		change_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ add = 1 }

	if = {
		limit = { NOT = { has_variable = $mod_id$__$setting_id$ } }

			always = no
			exists = flag:$mod_id$__$setting_id$
			exists = flag:$mod_id$__$tab_id$
			exists = flag:$mod_id$_$setting_id$_name
			exists = flag:$mod_id$_$setting_id$_desc
			exists = var:cmm_registered_setting_keys
			exists = var:cmm_setting_is_global_$mod_id$__$setting_id$
			exists = var:cmm_setting_is_slider_$mod_id$__$setting_id$

			exists = var:cmm_setting_owner_tab_key_$mod_id$__$setting_id$

			exists = var:$mod_id$__$setting_id$_name
			exists = var:$mod_id$__$setting_id$_desc
		}
	}
}

# Quantize a slider setting cache value to the nearest configured step.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_quantize_slider_setting = {
	if = {
		limit = { var:cmm_setting_is_global_$setting$ = 1 }

		set_variable = { name = cmm_tmp_slider_value value = global_var:$setting$ }
		change_variable = { name = cmm_tmp_slider_value subtract = var:cmm_setting_min_$setting$ }
		change_variable = { name = cmm_tmp_slider_value divide = var:cmm_setting_step_$setting$ }
		round_variable = {
			name = cmm_tmp_slider_value
			nearest = 1
		}
		change_variable = { name = cmm_tmp_slider_value multiply = var:cmm_setting_step_$setting$ }
		change_variable = { name = cmm_tmp_slider_value add = var:cmm_setting_min_$setting$ }

		if = {
			limit = { var:cmm_tmp_slider_value < var:cmm_setting_min_$setting$ }
			set_variable = { name = cmm_tmp_slider_value value = var:cmm_setting_min_$setting$ }
		}

		if = {
			limit = { var:cmm_tmp_slider_value > var:cmm_setting_max_$setting$ }
			set_variable = { name = cmm_tmp_slider_value value = var:cmm_setting_max_$setting$ }
		}

		set_global_variable = { name = $setting$ value = var:cmm_tmp_slider_value }
		remove_variable = cmm_tmp_slider_value

		cmm_sync_global_setting_cache = {
			setting = $setting$
		}
	}
	else = {
		set_variable = { name = cmm_tmp_slider_value value = var:$setting$ }
		change_variable = { name = cmm_tmp_slider_value subtract = var:cmm_setting_min_$setting$ }
		change_variable = { name = cmm_tmp_slider_value divide = var:cmm_setting_step_$setting$ }
		round_variable = {
			name = cmm_tmp_slider_value
			nearest = 1
		}
		change_variable = { name = cmm_tmp_slider_value multiply = var:cmm_setting_step_$setting$ }
		change_variable = { name = cmm_tmp_slider_value add = var:cmm_setting_min_$setting$ }

		if = {
			limit = { var:cmm_tmp_slider_value < var:cmm_setting_min_$setting$ }
			set_variable = { name = cmm_tmp_slider_value value = var:cmm_setting_min_$setting$ }
		}

		if = {
			limit = { var:cmm_tmp_slider_value > var:cmm_setting_max_$setting$ }
			set_variable = { name = cmm_tmp_slider_value value = var:cmm_setting_max_$setting$ }
		}

		set_variable = { name = $setting$ value = var:cmm_tmp_slider_value }
		remove_variable = cmm_tmp_slider_value
	}
}

		if = {
			limit = { NOT = { has_variable = cmm_tab_setting_count_$mod_id$__$tab_id$ } }
			set_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ value = 0 }
		}
		change_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ add = 1 }
	}

	# Default initialization; preserves existing save value.
	if = {
		limit = { NOT = { has_variable = $mod_id$__$setting_id$ } }
		set_variable = { name = $mod_id$__$setting_id$ value = $default_value$ }
	}

	# Clamp current value into configured range before quantizing to slider steps.
	if = {
		limit = { var:$mod_id$__$setting_id$ < $min_value$ }
		set_variable = { name = $mod_id$__$setting_id$ value = $min_value$ }
	}

	if = {
		limit = { var:$mod_id$__$setting_id$ > $max_value$ }
		set_variable = { name = $mod_id$__$setting_id$ value = $max_value$ }
	}

	cmm_quantize_slider_setting = {
		setting = $mod_id$__$setting_id$
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$__$setting_id$
			exists = flag:$mod_id$__$tab_id$
			exists = flag:$mod_id$_$setting_id$_name
			exists = flag:$mod_id$_$setting_id$_desc
			exists = var:cmm_registered_setting_keys
			exists = var:cmm_setting_is_global_$mod_id$__$setting_id$
			exists = var:cmm_setting_is_slider_$mod_id$__$setting_id$
			exists = var:cmm_setting_owner_mod_id_$mod_id$__$setting_id$
			exists = var:cmm_setting_owner_tab_key_$mod_id$__$setting_id$
			exists = var:cmm_setting_min_$mod_id$__$setting_id$
			exists = var:cmm_setting_max_$mod_id$__$setting_id$
			exists = var:cmm_setting_step_$mod_id$__$setting_id$
			exists = var:cmm_setting_slider_actual_step_count_$mod_id$__$setting_id$
			exists = var:cmm_setting_slider_visual_step_count_$mod_id$__$setting_id$
			exists = var:cmm_setting_slider_actual_last_index_$mod_id$__$setting_id$
			exists = var:cmm_setting_slider_visual_last_index_$mod_id$__$setting_id$
			exists = var:cmm_tab_setting_count_$mod_id$__$tab_id$
			exists = var:$mod_id$__$setting_id$_name
			exists = var:$mod_id$__$setting_id$_desc
		}
	}
}

# Register a dropdown setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - default_index (number): Initial selected option index.
# - option_count (number): Number of options (`>= 1`; indexed `0..option_count-1`).
cmm_register_dropdown_setting = {
	# Ensure owning tab exists before binding setting metadata.
	cmm_register_tab = {
		mod_id = $mod_id$
		tab_id = $tab_id$
	}

	set_variable = { name = cmm_setting_is_global_$mod_id$__$setting_id$ value = 0 }
	set_variable = { name = cmm_setting_is_slider_$mod_id$__$setting_id$ value = 0 }

	set_variable = { name = cmm_setting_dropdown_last_index_$mod_id$__$setting_id$ value = $option_count$ }
	change_variable = { name = cmm_setting_dropdown_last_index_$mod_id$__$setting_id$ add = -1 }

	# First-time registration path for this setting key.
	if = {
		limit = { NOT = { has_variable = cmm_registered_setting_$mod_id$__$setting_id$ } }
		set_variable = { name = cmm_registered_setting_$mod_id$__$setting_id$ value = yes }
		set_variable = { name = cmm_setting_owner_mod_id_$mod_id$__$setting_id$ value = flag:$mod_id$ }
		set_variable = { name = cmm_setting_owner_tab_key_$mod_id$__$setting_id$ value = flag:$mod_id$__$tab_id$ }
		set_variable = { name = $mod_id$__$setting_id$_name value = flag:$mod_id$_$setting_id$_name }
		set_variable = { name = $mod_id$__$setting_id$_desc value = flag:$mod_id$_$setting_id$_desc }

		# Global setting key list; used by right-panel datamodel iteration.
		add_to_variable_list = {
			name = cmm_registered_setting_keys
			target = flag:$mod_id$__$setting_id$
		}

		# Per-tab count; used for "no settings in this tab" UI state.
		if = {
			limit = { NOT = { has_variable = cmm_tab_setting_count_$mod_id$__$tab_id$ } }

		}
		change_variable = { name = cmm_tab_setting_count_$mod_id$__$tab_id$ add = 1 }
	}

	# Default initialization; preserves existing save value.
	if = {
		limit = { NOT = { has_variable = $mod_id$__$setting_id$ } }
		set_variable = { name = $mod_id$__$setting_id$ value = $default_index$ }
	}

	# Clamp current value into configured option range.
	if = {
		limit = { var:$mod_id$__$setting_id$ < 0 }
		set_variable = { name = $mod_id$__$setting_id$ value = 0 }
	}

	if = {
		limit = { var:$mod_id$__$setting_id$ > var:cmm_setting_dropdown_last_index_$mod_id$__$setting_id$ }
		set_variable = { name = $mod_id$__$setting_id$ value = var:cmm_setting_dropdown_last_index_$mod_id$__$setting_id$ }
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$__$setting_id$
			exists = flag:$mod_id$__$tab_id$
			exists = flag:$mod_id$_$setting_id$_name
			exists = flag:$mod_id$_$setting_id$_desc
			exists = var:cmm_registered_setting_keys
			exists = var:cmm_setting_is_global_$mod_id$__$setting_id$
			exists = var:cmm_setting_is_slider_$mod_id$__$setting_id$
			exists = var:cmm_setting_owner_mod_id_$mod_id$__$setting_id$
			exists = var:cmm_setting_owner_tab_key_$mod_id$__$setting_id$
			exists = var:cmm_setting_dropdown_count_$mod_id$__$setting_id$
			exists = var:cmm_setting_dropdown_last_index_$mod_id$__$setting_id$
			exists = var:cmm_tab_setting_count_$mod_id$__$tab_id$
			exists = var:$mod_id$__$setting_id$_name
			exists = var:$mod_id$__$setting_id$_desc
		}
	}
}

# Sync one global setting value into country cache for all human players.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_sync_global_setting_cache = {
	if = {
		limit = { has_global_variable = $setting$ }
		every_country = {

# Toggle a registered boolean setting value between `0` and `1`.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_toggle_bool_setting = {
	if = {
		limit = { var:cmm_setting_is_global_$setting$ = 1 }
		if = {
			limit = { global_var:$setting$ = 1 }
			set_global_variable = { name = $setting$ value = 0 }
		}
		else = {
			set_global_variable = { name = $setting$ value = 1 }
		}

		cmm_sync_global_setting_cache = {
			setting = $setting$
		}
	}
	else = {
		if = {
			limit = { var:$setting$ = 1 }
			set_variable = { name = $setting$ value = 0 }
		}
		else = {
			set_variable = { name = $setting$ value = 1 }
		}
	}
}

# Apply the pending dropdown selected index to a setting and clear the marker.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_apply_dropdown_change = {
	if = {
		limit = { has_variable = cmm_dropdown_index_to_apply }

		if = {
			limit = { var:cmm_setting_is_global_$setting$ = 1 }
			set_global_variable = { name = $setting$ value = var:cmm_dropdown_index_to_apply }

			if = {
				limit = { global_var:$setting$ < 0 }
				set_global_variable = { name = $setting$ value = 0 }
			}

			if = {
				limit = { global_var:$setting$ > var:cmm_setting_dropdown_last_index_$setting$ }
				set_global_variable = { name = $setting$ value = var:cmm_setting_dropdown_last_index_$setting$ }
			}

			cmm_sync_global_setting_cache = {
				setting = $setting$
			}
		}
		else = {
			set_variable = { name = $setting$ value = var:cmm_dropdown_index_to_apply }

			if = {
				limit = { var:$setting$ < 0 }
				set_variable = { name = $setting$ value = 0 }
			}

			if = {
				limit = { var:$setting$ > var:cmm_setting_dropdown_last_index_$setting$ }
				set_variable = { name = $setting$ value = var:cmm_setting_dropdown_last_index_$setting$ }
			}
		}

		remove_variable = cmm_dropdown_index_to_apply
	}

	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = var:cmm_dropdown_index_to_apply
			exists = var:cmm_setting_dropdown_last_index_$setting$
		}
	}
}

# Increase a numeric setting and clamp to its max value.

			limit = { var:$setting$ > var:cmm_setting_max_$setting$ }

		}
	}
}

# Set a slider setting directly from the pending visual-track index and clamp to range.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_set_slider_setting_from_visual_index = {
	set_variable = { name = cmm_tmp_slider_actual_index value = var:cmm_slider_index_to_apply }
	change_variable = { name = cmm_tmp_slider_actual_index multiply = var:cmm_setting_slider_actual_last_index_$setting$ }
	change_variable = { name = cmm_tmp_slider_actual_index divide = var:cmm_setting_slider_visual_last_index_$setting$ }
	round_variable = {
		name = cmm_tmp_slider_actual_index
		nearest = 1
	}

	set_variable = { name = cmm_tmp_slider_value value = var:cmm_setting_step_$setting$ }
	change_variable = { name = cmm_tmp_slider_value multiply = var:cmm_tmp_slider_actual_index }
	change_variable = { name = cmm_tmp_slider_value add = var:cmm_setting_min_$setting$ }

	if = {
		limit = { var:cmm_tmp_slider_value < var:cmm_setting_min_$setting$ }
		set_variable = { name = cmm_tmp_slider_value value = var:cmm_setting_min_$setting$ }
	}

	if = {
		limit = { var:cmm_tmp_slider_value > var:cmm_setting_max_$setting$ }
		set_variable = { name = cmm_tmp_slider_value value = var:cmm_setting_max_$setting$ }
	}

	if = {
		limit = { var:cmm_setting_is_global_$setting$ = 1 }
		set_global_variable = { name = $setting$ value = var:cmm_tmp_slider_value }

		cmm_sync_global_setting_cache = {
			setting = $setting$
		}
	}
	else = {
		set_variable = { name = $setting$ value = var:cmm_tmp_slider_value }
	}

	remove_variable = cmm_tmp_slider_actual_index
	remove_variable = cmm_tmp_slider_value
}

# Apply the pending numeric change marker to a setting and clear the marker.
# Scope: country
# Args:
# - setting (key): Composite setting key `<mod_id>__<setting_id>`.
cmm_apply_numeric_change = {
	if = {
		limit = { has_variable = cmm_numeric_change_to_apply }

		switch = {
			trigger = var:cmm_numeric_change_to_apply
			1 = { cmm_step_numeric_setting_up = { setting = $setting$ } }
			2 = { cmm_step_numeric_setting_up_5x = { setting = $setting$ } }
			3 = { cmm_set_numeric_setting_max = { setting = $setting$ } }
			-1 = { cmm_step_numeric_setting_down = { setting = $setting$ } }
			-2 = { cmm_step_numeric_setting_down_5x = { setting = $setting$ } }
			-3 = { cmm_set_numeric_setting_min = { setting = $setting$ } }
		}

		if = {
			limit = { var:cmm_setting_is_global_$setting$ = 1 }
			cmm_sync_global_setting_cache = {
				setting = $setting$
			}
		}

		remove_variable = cmm_numeric_change_to_apply
	}
}

# Apply the pending slider track selection or numeric step marker to a slider setting.

		}
	}
	else = {
		cmm_apply_numeric_change = {
			setting = $setting$
		}
	}
}
