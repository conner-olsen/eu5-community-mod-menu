# Register an ordered list setting for a mod (idempotent by mod_id + setting_id).
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Setting id unique within mod.
# - tab_id (key): Owner tab id within mod.
# - item_count (number): Number of ordered list items (`1..20`).
cmm_register_list_setting = {
	set_variable = { name = cmm_setting_is_global_$mod_id$__$setting_id$ value = 0 }
	set_variable = { name = cmm_setting_is_slider_$mod_id$__$setting_id$ value = 0 }
	set_variable = { name = cmm_setting_is_button_$mod_id$__$setting_id$ value = 0 }
	set_variable = { name = cmm_setting_list_count_$mod_id$__$setting_id$ value = $item_count$ }

	if = {
		limit = { var:cmm_setting_list_count_$mod_id$__$setting_id$ < 1 }
		set_variable = { name = cmm_setting_list_count_$mod_id$__$setting_id$ value = 1 }
	}

	if = {
		limit = { var:cmm_setting_list_count_$mod_id$__$setting_id$ > 20 }
		set_variable = { name = cmm_setting_list_count_$mod_id$__$setting_id$ value = 20 }
	}

	set_variable = {
		name = cmm_setting_list_last_index_$mod_id$__$setting_id$
		value = var:cmm_setting_list_count_$mod_id$__$setting_id$
	}
	change_variable = { name = cmm_setting_list_last_index_$mod_id$__$setting_id$ add = -1 }

	if = {
		limit = { NOT = { has_variable = cmm_list_field_count_$mod_id$__$setting_id$ } }
		set_variable = { name = cmm_list_field_count_$mod_id$__$setting_id$ value = 0 }
	}

	set_variable = {
		name = cmm_list_item_column_name_$mod_id$__$setting_id$
		value = flag:$mod_id$_$setting_id$_item_column_name
	}

	cmm_register_setting_metadata = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		tab_id = $tab_id$
		group_id = $setting_id$
	}

	cmm_initialize_list_setting_item_metadata = {
		mod_id = $mod_id$
		setting_id = $setting_id$
	}

	if = {
		limit = { NOT = { has_variable = cmm_list_initialized_$mod_id$__$setting_id$ } }

		cmm_initialize_list_setting_item_list = {
			mod_id = $mod_id$
			setting_id = $setting_id$
		}

		set_variable = { name = cmm_list_initialized_$mod_id$__$setting_id$ value = 1 }
	}

	cmm_suppress_list_setting_warnings = {
		mod_id = $mod_id$
		setting_id = $setting_id$
	}
}

# Register a bool field shared by every item in one ordered list setting.
# Scope: country
# Args:
# - mod_id (key): Owner mod id.
# - setting_id (key): Target list setting id within the mod.
# - field_id (key): Field id unique within the list setting.
# - default_value (1/0): Default checkbox state for new saves.
cmm_register_list_bool_field = {
	cmm_register_list_field_slot = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		field_id = $field_id$
	}

	if = {
		limit = {
			AND = {
				has_variable = cmm_setting_list_count_$mod_id$__$setting_id$
				has_variable = cmm_list_field_slot_$mod_id$__$setting_id$__$field_id$
			}
		}
		switch = {
			trigger = var:cmm_list_field_slot_$mod_id$__$setting_id$__$field_id$
			0 = { cmm_initialize_list_bool_field_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 0 default_value = $default_value$ } }
			1 = { cmm_initialize_list_bool_field_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 1 default_value = $default_value$ } }
			2 = { cmm_initialize_list_bool_field_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 2 default_value = $default_value$ } }
			3 = { cmm_initialize_list_bool_field_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 3 default_value = $default_value$ } }
			4 = { cmm_initialize_list_bool_field_slot = { mod_id = $mod_id$ setting_id = $setting_id$ field_id = $field_id$ slot = 4 default_value = $default_value$ } }
		}
	}

	cmm_suppress_list_bool_field_warnings = {
		mod_id = $mod_id$
		setting_id = $setting_id$
		field_id = $field_id$
	}
}
