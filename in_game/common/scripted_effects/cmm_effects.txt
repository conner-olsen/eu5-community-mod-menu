# Root scope: country (use the human player country when called from GUI/on_action).
#
# Public API:
# cmm_register_mod = {
#     mod_id = <required>
#     display_name_key = <required>
#     description_key = <required>
# }
#
# cmm_register_bool_setting = {
#     mod_id = <required>
#     setting_id = <required>
#     display_name_key = <required>
#     description_key = <required>
#     default_value = <optional, 0|1, default 0>
# }
#
# Setting change callback contract:
# - CMM calls "<mod_id>__<setting_id>_on_changed" immediately after a toggle.
# - Effects run in player country scope.

cmm_register_core_country = {
	cmm_register_mod = {
		mod_id = cmm_core
		display_name_key = CMM_CORE_MOD_NAME
		description_key = CMM_CORE_MOD_DESC
	}

	cmm_register_bool_setting = {
		mod_id = cmm_core
		setting_id = example_toggle
		display_name_key = CMM_CORE_BOOL_EXAMPLE_NAME
		description_key = CMM_CORE_BOOL_EXAMPLE_DESC
		default_value = 0
	}
}

cmm_register_mod = {
	if = {
		limit = { NOT = { has_variable = cmm_registered_mod_count } }
		set_variable = { name = cmm_registered_mod_count value = 0 }
	}

	if = {
		limit = { NOT = { has_variable = cmm_registered_mod_$mod_id$ } }
		change_variable = { name = cmm_registered_mod_count add = 1 }

		# First-seen order is stable and used as default list order.
		if = {
			limit = { NOT = { has_variable = cmm_mod_registration_order_$mod_id$ } }
			set_variable = {
				name = cmm_mod_registration_order_$mod_id$
				value = var:cmm_registered_mod_count
			}
		}

		add_to_variable_list = {
			name = cmm_registered_mod_ids
			target = flag:$mod_id$
		}
		add_to_variable_list = {
			name = cmm_registered_mod_display_name_keys
			target = flag:$display_name_key$
		}

		if = {
			limit = { var:cmm_registered_mod_count = 1 }
			set_variable = { name = cmm_first_registered_mod_id value = flag:$mod_id$ }
			set_variable = { name = cmm_first_registered_display_name_key value = flag:$display_name_key$ }
		}
	}

	set_variable = { name = cmm_registered_mod_$mod_id$ value = 1 }
	set_variable = { name = cmm_mod_active_$mod_id$ value = 1 }

	# Store localization metadata for runtime GUI reads.
	set_variable = { name = cmm_mod_display_name_key_$mod_id$ value = flag:$display_name_key$ }
	set_variable = { name = cmm_mod_description_key_$mod_id$ value = flag:$description_key$ }
	set_variable = { name = cmm_mod_description_key_by_display_name_key_$display_name_key$ value = flag:$description_key$ }

	# Benign error suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$
			exists = flag:$display_name_key$
			exists = flag:$description_key$
			exists = var:cmm_registered_mod_ids
			exists = var:cmm_registered_mod_display_name_keys
			exists = var:cmm_first_registered_mod_id
			exists = var:cmm_first_registered_display_name_key
			exists = var:cmm_registered_mod_$mod_id$
			exists = var:cmm_mod_active_$mod_id$
			exists = var:cmm_mod_display_name_key_$mod_id$
			exists = var:cmm_mod_description_key_$mod_id$
			exists = var:cmm_mod_description_key_by_display_name_key_$display_name_key$
		}
	}
}

cmm_unregister_mod = {
	if = {
		limit = { has_variable = cmm_registered_mod_$mod_id$ }

		remove_list_variable = {
			name = cmm_registered_mod_ids
			target = flag:$mod_id$
		}

		if = {
			limit = { has_variable = cmm_mod_display_name_key_$mod_id$ }
			remove_list_variable = {
				name = cmm_registered_mod_display_name_keys
				target = var:cmm_mod_display_name_key_$mod_id$
			}
		}

		remove_variable = cmm_registered_mod_$mod_id$
		remove_variable = cmm_mod_active_$mod_id$
		remove_variable = cmm_mod_display_name_key_$mod_id$
		remove_variable = cmm_mod_description_key_$mod_id$
		remove_variable = cmm_mod_registration_order_$mod_id$

		if = {
			limit = {
				has_variable = cmm_registered_mod_count
				var:cmm_registered_mod_count > 0
			}
			change_variable = { name = cmm_registered_mod_count subtract = 1 }
		}
	}

	remove_variable = cmm_mod_sort_priority_$mod_id$
}

cmm_set_mod_sort_priority = {
	set_variable = { name = cmm_mod_sort_priority_$mod_id$ value = $sort_priority$ }
}

cmm_register_bool_setting = {
	if = {
		limit = { NOT = { has_variable = cmm_registered_setting_$mod_id$__$setting_id$ } }
		set_variable = { name = cmm_registered_setting_$mod_id$__$setting_id$ value = 1 }

		add_to_variable_list = {
			name = cmm_registered_setting_keys
			target = flag:$mod_id$__$setting_id$
		}

		if = {
			limit = { NOT = { has_variable = cmm_mod_bool_setting_count_$mod_id$ } }
			set_variable = { name = cmm_mod_bool_setting_count_$mod_id$ value = 0 }
		}
		change_variable = { name = cmm_mod_bool_setting_count_$mod_id$ add = 1 }
	}

	set_variable = { name = cmm_setting_owner_mod_id_$mod_id$__$setting_id$ value = flag:$mod_id$ }

	if = {
		limit = { has_variable = cmm_mod_display_name_key_$mod_id$ }
		set_variable = {
			name = cmm_setting_owner_display_key_$mod_id$__$setting_id$
			value = var:cmm_mod_display_name_key_$mod_id$
		}
	}

	set_variable = { name = cmm_setting_label_key_$mod_id$__$setting_id$ value = flag:$display_name_key$ }
	set_variable = { name = cmm_setting_desc_key_$mod_id$__$setting_id$ value = flag:$description_key$ }

	if = {
		limit = {
			NOT = { has_variable = cmm_setting_bool_value_$mod_id$__$setting_id$ }
			$default_value$ = 1
		}
		set_variable = { name = cmm_setting_bool_value_$mod_id$__$setting_id$ value = 1 }
	}
	# Benign warning suppression
	if = {
		limit = {
			always = no
			exists = flag:$mod_id$__$setting_id$
			exists = flag:$display_name_key$
			exists = flag:$description_key$
			exists = var:cmm_registered_setting_keys
			exists = var:cmm_setting_owner_mod_id_$mod_id$__$setting_id$
			exists = var:cmm_setting_owner_display_key_$mod_id$__$setting_id$
			exists = var:cmm_setting_label_key_$mod_id$__$setting_id$
			exists = var:cmm_setting_desc_key_$mod_id$__$setting_id$
		}
	}
}

cmm_toggle_bool_setting = {
	if = {
		limit = { has_variable = cmm_setting_bool_value_$setting_key$ }
		remove_variable = cmm_setting_bool_value_$setting_key$
	}
	else = {
		set_variable = { name = cmm_setting_bool_value_$setting_key$ value = 1 }
	}
}

# Core example setting callback (triggered by "<setting_key>_on_changed" naming contract).
cmm_core__example_toggle_on_changed = {
	if = {
		limit = {
			cmm_is_bool_setting_enabled = {
				mod_id = cmm_core
				setting_id = example_toggle
			}
		}
		set_variable = { name = cmm_core_example_toggle_enabled value = 1 }
	}
	else = {
		remove_variable = cmm_core_example_toggle_enabled
	}
}
