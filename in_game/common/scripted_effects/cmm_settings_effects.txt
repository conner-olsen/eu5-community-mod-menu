# Register built-in CMM settings.
# Scope: country
# Args: none
cmm_register_menu_settings = {
	cmm_register_mod = {
		mod_id = cmm_core
	}

	cmm_register_global_button_setting = {
		mod_id = cmm_core
		setting_id = mark_host_country
		tab_id = general
		group_id = session
	}
}

# Mark the current country as the CMM host country.
# Scope: country
# Args: none
cmm_mark_host_country = {
	set_global_variable = { name = cmm_core_host_country value = this }
}
