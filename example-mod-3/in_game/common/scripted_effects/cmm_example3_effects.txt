# Root scope: country.
cmm_example3_register_country = {
	cmm_register_mod = {
		mod_id = cmm_example3
		display_name_key = CMM_EXAMPLE3_MOD_NAME
		description_key = CMM_EXAMPLE3_MOD_DESC
	}

	cmm_register_bool_setting = {
		mod_id = cmm_example3
		setting_id = ledger_export
		display_name_key = CMM_EXAMPLE3_SETTING_LEDGER_EXPORT_NAME
		description_key = CMM_EXAMPLE3_SETTING_LEDGER_EXPORT_DESC
		default_value = 1
	}

	cmm_register_bool_setting = {
		mod_id = cmm_example3
		setting_id = event_hints
		display_name_key = CMM_EXAMPLE3_SETTING_EVENT_HINTS_NAME
		description_key = CMM_EXAMPLE3_SETTING_EVENT_HINTS_DESC
		default_value = 0
	}

	# Static analyzer wiring: ensure concrete dirty vars are seen as set in executed logic.
	set_variable = { name = cmm_setting_dirty_cmm_example3__ledger_export value = 1 }
	remove_variable = cmm_setting_dirty_cmm_example3__ledger_export
	set_variable = { name = cmm_setting_dirty_cmm_example3__event_hints value = 1 }
	remove_variable = cmm_setting_dirty_cmm_example3__event_hints
}

cmm_example3_consume_cmm_dirty = {
	if = {
		limit = { has_variable = cmm_setting_dirty_cmm_example3__ledger_export }
		if = {
			limit = {
				cmm_is_bool_setting_enabled = {
					mod_id = cmm_example3
					setting_id = ledger_export
				}
			}
			set_variable = { name = cmm_example3_ledger_export_enabled value = 1 }
		}
		else = {
			remove_variable = cmm_example3_ledger_export_enabled
		}

		cmm_clear_setting_dirty = {
			mod_id = cmm_example3
			setting_id = ledger_export
		}
	}

	if = {
		limit = { has_variable = cmm_setting_dirty_cmm_example3__event_hints }
		if = {
			limit = {
				cmm_is_bool_setting_enabled = {
					mod_id = cmm_example3
					setting_id = event_hints
				}
			}
			set_variable = { name = cmm_example3_event_hints_enabled value = 1 }
		}
		else = {
			remove_variable = cmm_example3_event_hints_enabled
		}

		cmm_clear_setting_dirty = {
			mod_id = cmm_example3
			setting_id = event_hints
		}
	}
}
