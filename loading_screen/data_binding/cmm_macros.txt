macro = {
    description = "Country variable accessor"
    definition = "CMMVar(Name)"
    replace_with = "Player.MakeScope.GetVariable(Name)"
}

macro = {
    description = "Resolve a flag-backed variable to localized text with fallback"
    definition = "CMMVarFlagOr(Name,Fallback)"
    replace_with = "Select_CString(Player.MakeScope.GetVariable(Name).IsSet, Player.MakeScope.GetVariable(Name).GetFlagName, Fallback)"
}

macro = {
    description = "Resolve default tab key for a mod id with fallback"
    definition = "CMMDefaultTabOr(ModId,Fallback)"
    replace_with = "Select_CString(Player.MakeScope.GetVariable(Concatenate('cmm_mod_default_tab_', ModId)).IsSet, Player.MakeScope.GetVariable(Concatenate('cmm_mod_default_tab_', ModId)).GetFlagName, Fallback)"
}

macro = {
    description = "True when selected mod variable-system key exists"
    definition = "CMMSelectedModExists"
    replace_with = "GetVariableSystem.Exists('cmm_selected_mod')"
}

macro = {
    description = "True when selected tab variable-system key exists"
    definition = "CMMSelectedTabExists"
    replace_with = "GetVariableSystem.Exists('cmm_selected_tab')"
}

macro = {
    description = "Resolve currently selected mod name localization with fallback"
    definition = "CMMSelectedModNameOr(Fallback)"
    replace_with = "Select_CString(And(GetVariableSystem.Exists('cmm_selected_mod'), Player.MakeScope.GetVariable(Concatenate(GetVariableSystem.Get('cmm_selected_mod'), '_name')).IsSet), Player.MakeScope.GetVariable(Concatenate(GetVariableSystem.Get('cmm_selected_mod'), '_name')).GetFlagName, Fallback)"
}

macro = {
    description = "Resolve currently selected mod description localization with fallback"
    definition = "CMMSelectedModDescOr(Fallback)"
    replace_with = "Select_CString(And(GetVariableSystem.Exists('cmm_selected_mod'), Player.MakeScope.GetVariable(Concatenate(GetVariableSystem.Get('cmm_selected_mod'), '_desc')).IsSet), Player.MakeScope.GetVariable(Concatenate(GetVariableSystem.Get('cmm_selected_mod'), '_desc')).GetFlagName, Fallback)"
}

macro = {
    description = "Read tab owner mod id for a tab key"
    definition = "CMMTabOwnerModId(TabKey)"
    replace_with = "Player.MakeScope.GetVariable(Concatenate('cmm_tab_owner_mod_id_', TabKey)).GetFlagName"
}

macro = {
    description = "Read setting owner mod id for a setting key"
    definition = "CMMSettingOwnerModId(SettingKey)"
    replace_with = "Player.MakeScope.GetVariable(Concatenate('cmm_setting_owner_mod_id_', SettingKey)).GetFlagName"
}

macro = {
    description = "Read setting owner tab key for a setting key"
    definition = "CMMSettingOwnerTabKey(SettingKey)"
    replace_with = "Player.MakeScope.GetVariable(Concatenate('cmm_setting_owner_tab_key_', SettingKey)).GetFlagName"
}

macro = {
    description = "True when selected tab has a tab-setting-count variable"
    definition = "CMMSelectedTabCountIsSet"
    replace_with = "Player.MakeScope.GetVariable(Concatenate('cmm_tab_bool_setting_count_', GetVariableSystem.Get('cmm_selected_tab'))).IsSet"
}

macro = {
    description = "Selected tab setting count value"
    definition = "CMMSelectedTabCountValue"
    replace_with = "Player.MakeScope.GetVariable(Concatenate('cmm_tab_bool_setting_count_', GetVariableSystem.Get('cmm_selected_tab'))).GetValue"
}

macro = {
    description = "True when no mods are registered yet"
    definition = "CMMNoModsRegistered"
    replace_with = "Or(Not(CMMVar('cmm_registered_mod_count').IsSet), EqualTo_CFixedPoint(CMMVar('cmm_registered_mod_count').GetValue, '(CFixedPoint)0'))"
}

macro = {
    description = "True when at least one mod is registered"
    definition = "CMMHasRegisteredMods"
    replace_with = "And(CMMVar('cmm_registered_mod_count').IsSet, Not(EqualTo_CFixedPoint(CMMVar('cmm_registered_mod_count').GetValue, '(CFixedPoint)0')))"
}

macro = {
    description = "Filter match for one mod row against current search text"
    definition = "CMMModMatchesSearch(ModId)"
    replace_with = "Or4(Not(GetVariableSystem.Exists('cmm_mod_search_text')), StringIsEmpty(GetVariableSystem.Get('cmm_mod_search_text')), StringContains(CMMVarFlagOr(Concatenate(ModId, '_name'), ModId), GetVariableSystem.Get('cmm_mod_search_text')), StringContains(ModId, GetVariableSystem.Get('cmm_mod_search_text')))"
}

macro = {
    description = "Localized row label for a mod row with index fallback"
    definition = "CMMModRowLabel(ModId,RowIndex)"
    replace_with = "CMMVarFlagOr(Concatenate(ModId, '_name'), Select_CString(StringIsEmpty(ModId), Concatenate(Localize('CMM_MOD_LIST_ENTRY_PREFIX'), ToString_int32(Add_int32(RowIndex, '(int32)1'))), ModId))"
}

macro = {
    description = "Visibility rule for one setting row"
    definition = "CMMSettingRowVisible(SettingKey)"
    replace_with = "And3(And(CMMSelectedModExists, GetVariableSystem.HasValue('cmm_selected_mod', CMMSettingOwnerModId(SettingKey))), And(CMMSelectedTabExists, GetVariableSystem.HasValue('cmm_selected_tab', CMMSettingOwnerTabKey(SettingKey))), GetScriptedGui(Concatenate(SettingKey, '_on_changed')).IsShown(GuiScope.SetRoot(GetPlayer.MakeScope).End))"
}

macro = {
    description = "Execute setting on-changed scripted GUI callback"
    definition = "CMMExecuteSettingChanged(SettingKey)"
    replace_with = "GetScriptedGui(Concatenate(SettingKey, '_on_changed')).Execute(GuiScope.SetRoot(GetPlayer.MakeScope).End)"
}

macro = {
    description = "Checkbox checked state for numeric bool setting (1 = on)"
    definition = "CMMIsSettingEnabled(SettingKey)"
    replace_with = "EqualTo_CFixedPoint(CMMVar(SettingKey).GetValue, '(CFixedPoint)1')"
}

macro = {
    description = "Show empty-state text when selected tab has zero visible settings"
    definition = "CMMShowNoSettingsForSelectedTab"
    replace_with = "And(CMMSelectedModExists, Or3(Not(CMMSelectedTabExists), Not(CMMSelectedTabCountIsSet), EqualTo_CFixedPoint(CMMSelectedTabCountValue, '(CFixedPoint)0')))"
}
