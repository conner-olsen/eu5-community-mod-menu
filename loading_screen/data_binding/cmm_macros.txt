macro = {
    description = "Filter match for one mod row against current search text"
    definition = "CMMModMatchesSearch(ModId)"
    replace_with = "Or4(Not(GetVariableSystem.Exists('cmm_mod_search_text')), StringIsEmpty(GetVariableSystem.Get('cmm_mod_search_text')), StringContains(CMMVar(Concatenate(ModId, '_name')).GetFlagName, GetVariableSystem.Get('cmm_mod_search_text')), StringContains(ModId, GetVariableSystem.Get('cmm_mod_search_text')))"
}

macro = {
    description = "Visibility rule for one setting row"
    definition = "CMMSettingRowVisible(SettingKey)"
    replace_with = "And3(And(CMMSelectedModExists, GetVariableSystem.HasValue('cmm_selected_mod', CMMSettingOwnerModId(SettingKey))), And(CMMSelectedTabExists, GetVariableSystem.HasValue('cmm_selected_tab', CMMSettingOwnerTabKey(SettingKey))), GetScriptedGui(Concatenate(SettingKey, '_on_changed')).IsShown(GuiScope.SetRoot(GetPlayer.MakeScope).End))"
}

macro = {
    description = "Execute setting on-changed scripted GUI callback"
    definition = "CMMExecuteSettingChanged(SettingKey)"
    replace_with = "GetScriptedGui(Concatenate(SettingKey, '_on_changed')).Execute(GuiScope.SetRoot(GetPlayer.MakeScope).End)"
}

macro = {
    description = "Checkbox checked state for numeric bool setting (1 = on)"
    definition = "CMMIsSettingEnabled(SettingKey)"
    replace_with = "EqualTo_CFixedPoint(CMMVar(SettingKey).GetValue, '(CFixedPoint)1')"
}

macro = {
    description = "Show empty-state text when selected tab has zero visible settings"
    definition = "CMMShowNoSettingsForSelectedTab"
    replace_with = "And(CMMSelectedModExists, Or3(Not(CMMSelectedTabExists), Not(CMMSelectedTabCountIsSet), EqualTo_CFixedPoint(CMMSelectedTabCountValue, '(CFixedPoint)0')))"
}

macro = {
    description = "Command string to toggle CMM window open/close"
    definition = "CMMWindowToggleCommand"
    replace_with = "Select_CString(GetVariableSystem.Exists('cmm_window_open'), 'gui.ClearWidgets cmm_window', 'gui.createwidget gui/cmm_window.gui cmm_window')"
}
