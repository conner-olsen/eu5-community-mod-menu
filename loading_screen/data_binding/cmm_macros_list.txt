# CMM GUI Macros: Ordered Lists

# Args: ItemKey (string)
# Returns: string
macro = {
    description = "Owner setting key for one ordered list item key"
    definition = "CMMListItemOwnerSettingKey(ItemKey)"
    replace_with = "CMMMetadataFlag('cmm_list_item_owner_setting_', ItemKey)"
}

# Args: ItemKey (string)
# Returns: boolean
macro = {
    description = "True when local player can edit the owner setting for one ordered list item"
    definition = "CMMCanEditListItemSetting(ItemKey)"
    replace_with = "CMMCanEditSetting(CMMListItemOwnerSettingKey(ItemKey))"
}

# Args: ItemKey (string)
# Returns: none
macro = {
    description = "Execute the owner setting callback for one ordered list item"
    definition = "CMMExecuteListItemSettingChanged(ItemKey)"
    replace_with = "CMMExecuteSettingChanged(CMMListItemOwnerSettingKey(ItemKey))"
}

# Args: SettingKey (string)
# Returns: int32
macro = {
    description = "Ordered list field count for one setting"
    definition = "CMMListFieldCount(SettingKey)"
    replace_with = "CMMMetadataInt('cmm_list_field_count_', SettingKey)"
}

# Args: SettingKey (string), SlotKey (string), Suffix (string)
# Returns: variable
macro = {
    description = "Ordered list field metadata variable accessor"
    definition = "CMMListFieldMetadataVar(SettingKey,SlotKey,Suffix)"
    replace_with = "CMMVar(Concatenate(Concatenate('cmm_list_field_', SlotKey), Concatenate(Suffix, SettingKey)))"
}

# Args: SettingKey (string), SlotKey (string), Suffix (string)
# Returns: string
macro = {
    description = "Ordered list field metadata flag-name accessor"
    definition = "CMMListFieldMetadataFlag(SettingKey,SlotKey,Suffix)"
    replace_with = "CMMListFieldMetadataVar(SettingKey, SlotKey, Suffix).GetFlagName"
}

# Args: SettingKey (string), SlotKey (string), Suffix (string)
# Returns: value
macro = {
    description = "Ordered list field metadata value accessor"
    definition = "CMMListFieldMetadataValue(SettingKey,SlotKey,Suffix)"
    replace_with = "CMMListFieldMetadataVar(SettingKey, SlotKey, Suffix).GetValue"
}

# Args: SettingKey (string), SlotKey (string), Suffix (string)
# Returns: int32
macro = {
    description = "Ordered list field metadata int accessor"
    definition = "CMMListFieldMetadataInt(SettingKey,SlotKey,Suffix)"
    replace_with = "FixedPointToInt(CMMListFieldMetadataValue(SettingKey, SlotKey, Suffix))"
}

# Args: ItemKey (string), SlotKey (string)
# Returns: variable
macro = {
    description = "Ordered list item field variable accessor"
    definition = "CMMListItemFieldVar(ItemKey,SlotKey)"
    replace_with = "CMMVar(Concatenate(ItemKey, Concatenate('_field_', SlotKey)))"
}

# Args: ItemKey (string), SlotKey (string)
# Returns: value
macro = {
    description = "Ordered list item field value accessor"
    definition = "CMMListItemFieldValue(ItemKey,SlotKey)"
    replace_with = "CMMListItemFieldVar(ItemKey, SlotKey).GetValue"
}

# Args: ItemKey (string), SlotKey (string)
# Returns: int32
macro = {
    description = "Ordered list item field int accessor"
    definition = "CMMListItemFieldInt(ItemKey,SlotKey)"
    replace_with = "FixedPointToInt(CMMListItemFieldValue(ItemKey, SlotKey))"
}

# Args: SettingKey (string), SlotKey (string)
# Returns: boolean
macro = {
    description = "True when one ordered list field slot is bool"
    definition = "CMMListFieldIsBool(SettingKey,SlotKey)"
    replace_with = "EqualTo_CFixedPoint(CMMListFieldMetadataValue(SettingKey, SlotKey, '_type_'), '(CFixedPoint)1')"
}

# Args: SettingKey (string), SlotKey (string)
# Returns: boolean
macro = {
    description = "True when one ordered list field slot is dropdown"
    definition = "CMMListFieldIsDropdown(SettingKey,SlotKey)"
    replace_with = "EqualTo_CFixedPoint(CMMListFieldMetadataValue(SettingKey, SlotKey, '_type_'), '(CFixedPoint)2')"
}

# Args: SettingKey (string), SlotKey (string)
# Returns: boolean
macro = {
    description = "True when one ordered list field slot is numeric"
    definition = "CMMListFieldIsNumeric(SettingKey,SlotKey)"
    replace_with = "EqualTo_CFixedPoint(CMMListFieldMetadataValue(SettingKey, SlotKey, '_type_'), '(CFixedPoint)3')"
}
