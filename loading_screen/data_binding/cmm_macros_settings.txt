# CMM GUI Macros: Settings

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "True when a setting key is registered as global"
    definition = "CMMSettingIsGlobal(SettingKey)"
    replace_with = "CMMMetadataEqualsOne('cmm_setting_is_global_', SettingKey)"
}

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "True when a setting key is registered as numeric type"
    definition = "CMMSettingIsNumeric(SettingKey)"
    replace_with = "And(CMMMetadataIsSet('cmm_setting_step_', SettingKey), Not(CMMSettingIsSlider(SettingKey)))"
}

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "True when a setting key is registered as bool type"
    definition = "CMMSettingIsBool(SettingKey)"
    replace_with = "And(And4(Not(CMMSettingIsNumeric(SettingKey)), Not(CMMSettingIsDropdown(SettingKey)), Not(CMMSettingIsSlider(SettingKey)), Not(CMMSettingIsText(SettingKey))), And(Not(CMMSettingIsButton(SettingKey)), Not(CMMSettingIsList(SettingKey))))"
}

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "True when local player can edit this setting"
    definition = "CMMCanEditSetting(SettingKey)"
    replace_with = "Or(Not(CMMSettingIsGlobal(SettingKey)), And(Or(EqualTo_string(SettingKey, 'cmm_core__enable_host_only_tools'), And(GetGlobalVariable('cmm_core__enable_host_only_tools').IsSet, CMMValueEqualsOne(GetGlobalVariable('cmm_core__enable_host_only_tools').GetValue))), Or(Not(GameIsMultiplayer), IsHost)))"
}

# Args: SettingKey (string)
# Returns: value
macro = {
    description = "Read current setting value from country or global scope"
    definition = "CMMSettingValue(SettingKey)"
    replace_with = "Select_CFixedPoint(CMMSettingIsGlobal(SettingKey), GetGlobalVariable(SettingKey).GetValue, CMMVar(SettingKey).GetValue)"
}

# Args: SettingKey (string)
# Returns: boolean
macro = {
    description = "Visibility rule for one setting row"
    definition = "CMMSettingRowVisible(SettingKey)"
    replace_with = "And(CMMMatchesSelectedModAndTab(CMMMetadataFlag('cmm_setting_owner_mod_id_', SettingKey), CMMMetadataFlag('cmm_setting_owner_tab_key_', SettingKey)), Or(CMMSettingIsText(SettingKey), CMMGuiIsShown(Concatenate(SettingKey, '_on_changed'))))"
}

}
